import ReactDOMServer from "react-dom/server";

export const Button = ({ children, color }) => (
  <span
    href="#"
    style={{
      backgroundColor: color,
      borderRadius: "20px",
      color: "#fff",
      padding: "10px",
      width: "200px",
    }}
    onClick={() => {
      alert(`Generated string markup\n
${generateStaticMarkup()}`);
    }}
  >
    {children}
  </span>
);

export const generateStaticMarkup = () =>
  ReactDOMServer.renderToStaticMarkup(
    <span
      href="#"
      style={{
        backgroundColor: "aqua",
        borderRadius: "20px",
        color: "#fff",
        padding: "10px",
        width: "200px",
      }}
    >
      hello
    </span>
  );
console.log(generateStaticMarkup());
